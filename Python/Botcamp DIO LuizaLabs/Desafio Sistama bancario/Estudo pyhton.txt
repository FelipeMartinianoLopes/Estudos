def depositar(saldo, valor, extrato, /):
    """
    Realiza um depósito na conta do usuário.
    
    Args (positional only):
        saldo (float): Saldo atual da conta
        valor (float): Valor a ser depositado
        extrato (str): Histórico de transações
    
    Returns:
        tuple: Saldo atualizado e extrato atualizado
    """
    if valor > 0:
        saldo += valor
        extrato += f"Depósito: R$ {valor:.2f}\n"
        print(f"Depósito de R$ {valor:.2f} realizado com sucesso!")
    else:
        print("Operação falhou! O valor informado é inválido.")
    
    return saldo, extrato


def sacar(*, saldo, valor, extrato, limite, numero_saques, limite_saques):
    """
    Realiza um saque na conta do usuário.
    
    Args (keyword only):
        saldo (float): Saldo atual da conta
        valor (float): Valor a ser sacado
        extrato (str): Histórico de transações
        limite (float): Limite máximo por saque
        numero_saques (int): Número de saques realizados hoje
        limite_saques (int): Limite máximo de saques por dia
    
    Returns:
        tuple: Saldo atualizado, extrato atualizado e número de saques atualizado
    """
    excedeu_saldo = valor > saldo
    excedeu_limite = valor > limite
    excedeu_saques = numero_saques >= limite_saques

    if excedeu_saldo:
        print("Operação falhou! Você não tem saldo suficiente.")
    elif excedeu_limite:
        print("Operação falhou! O valor do saque excede o limite.")
    elif excedeu_saques:
        print("Operação falhou! Número máximo de saques excedido.")
    elif valor > 0:
        saldo -= valor
        extrato += f"Saque: R$ {valor:.2f}\n"
        numero_saques += 1
        print(f"Saque de R$ {valor:.2f} realizado com sucesso!")
    else:
        print("Operação falhou! O valor informado é inválido.")
    
    return saldo, extrato, numero_saques


def exibir_extrato(saldo, /, *, extrato):
    """
    Exibe o extrato bancário do usuário.
    
    Args:
        saldo (positional only): Saldo atual da conta
        extrato (keyword only): Histórico de transações
    """
    print("\n" + "=" * 40 + " EXTRATO " + "=" * 40)
    print("Não foram realizadas movimentações." if not extrato else extrato)
    print(f"\nSaldo: R$ {saldo:.2f}")
    print("=" * 89)


def criar_usuario(usuarios):
    """
    Cria um novo usuário (cliente do banco).
    
    Args:
        usuarios (list): Lista de usuários existentes
    
    Returns:
        list: Lista de usuários atualizada
    """
    cpf = input("Informe o CPF (somente números): ").strip()
    
    usuario_existente = filtrar_usuario(cpf, usuarios)
    
    if usuario_existente:
        print("Erro: Já existe usuário com esse CPF!")
        return usuarios
    
    nome = input("Informe o nome completo: ").strip()
    data_nascimento = input("Informe a data de nascimento (dd-mm-aaaa): ").strip()
    endereco = input("Informe o endereço (logradouro, nro - bairro - cidade/sigla estado): ").strip()
    
    cpf_numeros = ''.join(filter(str.isdigit, cpf))
    
    novo_usuario = {
        "nome": nome,
        "data_nascimento": data_nascimento,
        "cpf": cpf_numeros,
        "endereco": endereco
    }
    
    usuarios.append(novo_usuario)
    print("Usuário criado com sucesso!")
    return usuarios


def filtrar_usuario(cpf, usuarios):
    """
    Filtra usuário por CPF.
    
    Args:
        cpf (str): CPF do usuário
        usuarios (list): Lista de usuários
    
    Returns:
        dict: Usuário encontrado ou None
    """
    cpf_numeros = ''.join(filter(str.isdigit, cpf))
    for usuario in usuarios:
        if usuario["cpf"] == cpf_numeros:
            return usuario
    return None


def criar_conta_corrente(agencia, numero_conta, usuarios):
    """
    Cria uma nova conta corrente vinculada a um usuário.
    
    Args:
        agencia (str): Número da agência
        numero_conta (int): Número da conta
        usuarios (list): Lista de usuários
    
    Returns:
        dict: Nova conta criada ou None se usuário não encontrado
    """
    cpf = input("Informe o CPF do usuário: ").strip()
    usuario = filtrar_usuario(cpf, usuarios)
    
    if not usuario:
        print("Erro: Usuário não encontrado! Crie o usuário primeiro.")
        return None
    
    nova_conta = {
        "agencia": agencia,
        "numero_conta": numero_conta,
        "usuario": usuario
    }
    
    print(f"Conta corrente {numero_conta} criada com sucesso para {usuario['nome']}!")
    return nova_conta


def listar_contas(contas):
    """
    Lista todas as contas cadastradas.
    
    Args:
        contas (list): Lista de contas
    """
    if not contas:
        print("\nNenhuma conta cadastrada.")
        return
    
    print("\n" + "=" * 60)
    print("LISTA DE CONTAS CADASTRADAS")
    print("=" * 60)
    
    for conta in contas:
        print(f"Agência: {conta['agencia']}")
        print(f"C/C: {conta['numero_conta']}")
        print(f"Titular: {conta['usuario']['nome']}")
        print(f"CPF: {conta['usuario']['cpf']}")
        print("-" * 60)


def main():
    """
    Função principal que executa o sistema bancário.
    """
    saldo = 0
    limite = 500
    extrato = ""
    numero_saques = 0
    LIMITE_SAQUES = 3
    
    usuarios = []
    contas = []
    
    AGENCIA = "0001"
    numero_conta = 1
    
    menu_principal = """
    [d] Depositar
    [s] Sacar
    [e] Extrato
    [u] Criar Usuário
    [c] Criar Conta Corrente
    [l] Listar Contas
    [q] Sair

    => """

    while True:
        opcao = input(menu_principal).strip().lower()

        if opcao == "d":
            valor = float(input("Informe o valor do depósito: "))
            saldo, extrato = depositar(saldo, valor, extrato)

        elif opcao == "s":
            valor = float(input("Informe o valor do saque: "))
            saldo, extrato, numero_saques = sacar(
                saldo=saldo,
                valor=valor,
                extrato=extrato,
                limite=limite,
                numero_saques=numero_saques,
                limite_saques=LIMITE_SAQUES
            )

        elif opcao == "e":
            exibir_extrato(saldo, extrato=extrato)

        elif opcao == "u":
            usuarios = criar_usuario(usuarios)

        elif opcao == "c":
            nova_conta = criar_conta_corrente(AGENCIA, numero_conta, usuarios)
            if nova_conta:
                contas.append(nova_conta)
                numero_conta += 1  # Incrementa para próxima conta

        elif opcao == "l":
            listar_contas(contas)

        elif opcao == "q":
            print("Saindo do sistema...")
            break

        else:
            print("Operação inválida, por favor selecione novamente a operação desejada.")


if __name__ == "__main__":
    main()
